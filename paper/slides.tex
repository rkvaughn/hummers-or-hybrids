% -- Section 1: Executive Summary ---------------------------------------------
\section{Overview}

\begin{frame}{Project Motivation}
  \begin{columns}[T]
    \begin{column}{0.52\textwidth}
      \small
      \textbf{Kahn (2007) established a clean empirical fact:}
      \begin{itemize}
        \item Green Party registration predicts hybrid ownership, lower gas use,
              and transit commuting --- even controlling for income and education
        \item Environmental ideology shows up in revealed transportation choices
              at the community level, not just in surveys
      \end{itemize}
      \textbf{18 years later, things have changed:}
      \begin{itemize}
        \item The Prius has given way to Tesla as the dominant green status vehicle
        \item Non-Tesla EVs now span a wide range of price points
        \item \textbf{Tesla is Elon Musk's} --- a question Kahn never had to ask
      \end{itemize}
    \end{column}
    \begin{column}{0.45\textwidth}
      \begin{block}{The questions we ask}
        \begin{enumerate}\small
          \item Does climate ideology still predict green transportation in 2023 CA?
          \item How does Tesla compare to non-Tesla EVs?
          \item \textbf{The Elon Effect:} has the ideology--Tesla link shifted post-Musk?
        \end{enumerate}
      \end{block}
      \begin{block}{Why it matters}
        \small If brand politics decouple green consumers from the leading EV
        maker, who fills the gap? Implications for EV diffusion, climate policy,
        and the limits of consumer activism.
      \end{block}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Key Findings}
  \begin{enumerate}
    \item \textbf{Kahn's finding survives.} Climate ideology predicts lower-carbon commuting
    (+1.09 pp transit, $-$1.84 pp drive-alone) and higher EV ownership (+5.0\% Tesla,
    +6.1\% non-Tesla) in 2023 California.

    \vspace{0.3em}
    \item \textbf{Non-Tesla EVs have democratized.} The ideology--non-Tesla link weakened
    2020--2022 as affordable models spread into lower-ideology communities. By 2023--2024
    the link largely recovered --- EVs are diffusing across the ideology spectrum.

    \vspace{0.3em}
    \item \textbf{The Elon Effect: present, but not yet cleanly identified.}
    \begin{itemize}\small
      \item Stock-based event study: no decline through 2024 --- but methodologically fragile
            (CEC data are accumulated stock, not new sales)
      \item \textbf{First-difference results:} high-ideology tracts have added fewer Teslas per
            year since 2019, while non-Tesla BEVs recover after 2021
      \item The slowdown predates Musk's Twitter acquisition --- saturation and competition
            are confounded with any brand-politics effect
    \end{itemize}

    \vspace{0.3em}
    \item \textbf{Spatial autocorrelation is strong but results hold.}
    Moran's I = 0.58; SAR $\hat{\rho}$ = 0.78; ideology coefficients survive spatial correction.
  \end{enumerate}
\end{frame}

      \begin{itemize}\small
        \item Tract $\times$ year panel, 2018--2024

        \item Full two-way FE infeasible --- ideology is time-invariant

            \item \textit{Stock-based} (Script 09): $\log(\text{EV}+1)$
            \item \textit{FD} (Script 09b): $\Delta\log(\text{EV}+1)$

\begin{frame}{A Methodological Issue: Stock vs.\ Flow}
  \begin{columns}[T]
    \begin{column}{0.60\textwidth}
      \small
      \textbf{The problem.}
      CEC data are December 31 \textit{stock} snapshots --- not new sales.
      A Tesla purchased in 2017 still appears in the 2024 file.

      \vspace{0.2em}
      With roughly 18\% annual EV fleet turnover, only about
      \textbf{33\% of the 2024 Tesla stock} reflects post-October 2022
      purchases. Even a real slowdown in high-ideology tracts would move
      the stock by a few percent --- within noise.

      \vspace{0.2em}
      \textbf{The fix: first-difference the outcome.}
      \[
        \Delta\log(\text{Tesla}+1)_{it} =
          \log(\text{Tesla}+1)_{it} - \log(\text{Tesla}+1)_{i,t-1}
      \]
      This approximates net new registrations
      (new purchases minus retirements and transfers).
      Tract FE are absorbed by first-differencing.
    \end{column}
    \begin{column}{0.37\textwidth}
      \begin{block}{The stock spec is not wrong for all purposes}
        \small Cross-section and panel regressions (``do greener tracts have
        more EVs?'') use stock correctly. The issue is specific to the
        \textit{event study}, which needs to detect changes in purchasing
        behavior.
      \end{block}
      \vspace{0.4em}
      \begin{block}{One mitigating observation}
        \small The non-Tesla BEV series dips in 2021--2022 in the stock
        event study, showing the spec is sensitive to some rate changes.
        It is attenuated, not completely blind.
      \end{block}
    \end{column}
  \end{columns}
\end{frame}

  Vertical lines: Twitter acquisition (Oct 2022) and DOGE/Trump role (Nov 2024).}

\begin{frame}{The Elon Effect: Revised Interpretation}
  \begin{columns}[T]
    \begin{column}{0.55\textwidth}
      \small
      \textbf{\textcolor{elonred}{Tesla (red):}} Negative in all post-2019 years
      ($-$0.017 to $-$0.032 per SD ideology).
      High-ideology tracts add fewer net new Teslas each year. Robust to county clustering.

      \vspace{0.15em}
      \textbf{\textcolor{evblue}{Non-Tesla BEV (blue):}} Negative early
      ($\approx$$-$0.025 in 2020), converges toward zero by 2023,
      diverging from Tesla after 2021.

      \vspace{0.15em}
      \textbf{The complication:} Tesla slowdown begins \textbf{2020} ---
      two years pre-Twitter acquisition. A clean Elon Effect attribution
      requires monthly new-sales data.
    \end{column}
    \begin{column}{0.42\textwidth}
      \begin{block}{Three candidate explanations}
        \begin{enumerate}\small
          \item \textbf{Saturation:} high-ideology tracts mostly bought
                Tesla before 2019; growth slowed mechanically
          \item \textbf{Substitution:} greener buyers shifted to
                Bolt/Ioniq/Rivian as alternatives matured
          \item \textbf{Elon Effect:} brand disutility post-2022
                accelerated an existing trend
        \end{enumerate}
      \end{block}
      \vspace{0.05em}
      \begin{alertblock}{The honest bottom line}
        \small High-ideology tracts have added Teslas more slowly
        since 2019. Isolating Musk's politics requires monthly data.
      \end{alertblock}
    \end{column}
  \end{columns}
\end{frame}

% -- Section 5: Conclusions ---------------------------------------------------
\section{Conclusions}

\begin{frame}{Conclusions}
  \begin{enumerate}
    \item \textbf{Kahn's finding survives into the EV era.}
    Climate ideology robustly predicts lower-carbon commuting and higher EV ownership
    in 2023 California. The green signal is alive.

    \vspace{0.4em}
    \item \textbf{Non-Tesla EVs have democratized.}
    The ideology--non-Tesla link weakened 2020--2022 as affordable models spread
    beyond high-ideology communities. The technology is outrunning the brand signal.

    \vspace{0.4em}
    \item \textbf{The Elon Effect: suggestive but not yet pinned down.}
    The stock-based null was methodologically fragile. First-difference results show
    high-ideology tracts adding fewer Teslas per year since 2019, while non-Tesla
    BEVs recover. The pattern predates Musk's political turn; 2025 CEC data is
    needed for a cleaner test.

    \vspace{0.4em}
    \item \textbf{Spatial autocorrelation is strong but does not reverse results.}
    Moran's I = 0.58 (transit); SAR $\hat{\rho}$ = 0.78. Ideology coefficients
    survive spatial correction.

    \vspace{0.4em}
    \item \textbf{All main results robust} to three alternative ideology specifications
    (YCOM-only, no-YCOM, Prop 30 alone).
  \end{enumerate}
\end{frame}

\begin{frame}{Next Steps}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \textbf{High priority:}
      \begin{enumerate}\small
        \item \textbf{2025 CEC data} --- re-run the FD event study with the
        DOGE/Trump window fully in sample; the definitive Elon Effect test
        \item \textbf{AFDC EV charger density} --- acquire from DOE AFDC API;
        ideology may proxy for Supercharger access, biasing coefficients upward
        \item \textbf{Housing type control} --- add \texttt{pct\_owner\_occupied}
        (ACS B25003) to absorb the EV charging constraint from multi-unit buildings
        \item \textbf{Time-varying ACS controls} --- use 2019 ACS vintage for
        2018--2020 panel years
      \end{enumerate}
    \end{column}
    \begin{column}{0.48\textwidth}
      \textbf{Extensions:}
      \begin{enumerate}\small
        \setcounter{enumi}{4}
        \item \textbf{National replication} --- extend beyond California
        using YCOM (which is national) and state DMV-equivalent data
        \item \textbf{Brand-level heterogeneity} --- decompose non-Tesla EVs
        by brand (Bolt vs.\ Rivian vs.\ Ioniq)
        \item \textbf{Low-MPG truck filter} --- restrict light trucks to
        below 20 MPG to sharpen the Hummer comparison
        \item \textbf{Publish} --- fill GitHub link, post to Substack,
        submit to NBER EEE working paper series
      \end{enumerate}
    \end{column}

\end{frame}
